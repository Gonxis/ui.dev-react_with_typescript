    In short, must specify the generic type for the useReducer function manually to help Typescript understands what the types are like this:

    ```ts
    <React.Reducer<FetchState<T>, FetchActionType<T>>>
    ```

    ```ts
    import * as React from "react";

    interface DadJokeResponse {
    id: string;
    joke: string;
    status: 200;
    }
    const JOKE_URL = "https://icanhazdadjoke.com/";

    type FetchStateType = "loading" | "data" | "error";

    interface FetchState<T> {
    state: FetchStateType;
    data: null | T;
    error: null | Error;
    }

    interface LoadingAction {
    type: "loading";
    }

    interface DataAction<T> {
    type: "data";
    data: T;
    }

    interface ErrorAction {
    type: "error";
    error: Error;
    }

    type FetchActionType<T> = LoadingAction | DataAction<T> | ErrorAction;

    function fetchReducer<T>(
    state: FetchState<T>,
    action: FetchActionType<T>
    ): FetchState<T> {
    //  Implement your reducer here.
    switch (action.type) {
        case "loading": {
        return { state: action.type, data: null, error: null };
        }
        case "data": {
        return { ...state, state: action.type, data: action.data };
        }
        case "error": {
        return { ...state, state: action.type, error: action.error };
        }
    }
    return state;
    }

    const initial: FetchState<DadJokeResponse> = {
    state: "loading",
    data: null,
    error: null
    };

    function useFetch<T>(initial: FetchState<T>, url: string): FetchState<T> {
    const [state, dispatch] = React.useReducer<React.Reducer<FetchState<T>, FetchActionType<T>>>(fetchReducer, initial);

    React.useEffect(() => {
        async function performFetch() {
        try {
            const response = await fetch(url, {
            headers: {
                accept: "application/json"
            }
            });
            const data = await response.json();
            dispatch({ type: "data", data });
        } catch (error) {
            dispatch({ type: "error", error });
        }
        }
        dispatch({ type: "loading" });
        performFetch();
    }, [url]);
    return state;
    }

    export default function App() {
    const { state, data, error } = useFetch<DadJokeResponse>(initial, JOKE_URL);
    if (state === "loading") return <div>Loading...</div>;
    if (state === "error") return <div>Error: {error?.message}</div>;
    if (state === "data") return <div>{data?.joke}</div>;
    throw new Error("This should never happen.");
    }
    ```

    /**
    * The App component uses a custom useFetch hook which
    * tracks its state with useReducer.

    * assign types to its parameters.
    *
    * Don't forget: Reducers are much easier to implement
    * in TypeScript if you use a discriminating union for the
    * Action type.
    *
    * Add any other type annotations where necessary, By the time
    * you are done, the red squiggles should be gone.
    *
    * Bonus points: Try making useFetch into a generic function so
    * it can be used with more than just the dad joke API.
    */

    interface DadJokeResponse {
    id: string;
    joke: string;
    status: 200;
    }
    const JOKE_URL = "https://icanhazdadjoke.com/";

    interface FetchState<T> {
    state: "loading" | "error" | "data";
    data: null | T;
    error: null | Error;
    }
    interface FetchDataAction<T> {
    type: "data";
    data: T;
    }
    interface FetchErrorAction {
    type: "error";
    error: Error;
    }
    interface FetchLoadingAction {
    type: "loading";
    }
    type FetchActions<T> = FetchDataAction<T> | FetchErrorAction | FetchLoadingAction;

    function fetchReducer<T>(state: FetchState<T>, action: FetchActions<T>): FetchState<T> {
    //  Implement your reducer here.
    switch (action.type) {
        case "data":
        return { state: "data", data: action.data, error: null };
        case "error":
        return { state: "error", data: null, error: action.error };
        case "loading":
        return { state: "loading", data: null, error: null };
    }
    return state;
    }

    function useFetch<T>(url: string): FetchState<T> {
    type ReducerType = (state: FetchState<T>, action: FetchActions<T>) => FetchState<T>;
    const [state, dispatch] = React.useReducer<ReducerType>(fetchReducer, {
        state: "loading",
        data: null, 
        error: null
    });

    React.useEffect(() => {
        async function performFetch() {
        try {
            const response = await fetch(url, {
            headers: {
                accept: "application/json"
            }
            });
            const data: T = await response.json();
            dispatch({ type: "data", data });
        } catch (error) {
            if (error instanceof Error) {
            dispatch({ type: "error", error });
            }
        }
        }
        dispatch({ type: "loading" });
        performFetch();
    }, [url]);

    return state;
    }

    export default function App() {
    const { state, data, error } = useFetch<DadJokeResponse>(JOKE_URL);
    if (state === "loading") return <div>Loading...</div>;
    if (state === "error") return <div>Error: {error?.message}</div>;
    if (state === "data") return <div>{data?.joke}</div>;
    throw new Error("This should never happen.");
    }
